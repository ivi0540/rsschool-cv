Task: given an array whose elements can be natural numbers or a new array with natural numbers, you need to find the sum of all numbers in the array. 

let arr1 = [1, [2, 7, 8], [3, 4], [5, [6, 7]]];
function getSum(arr) {
    let sum = 0;
    for (let elem of arr) {
        if (Number(elem)) {
            sum += elem;
        } else {
            sum += getSum(elem);
        };
    };
    return sum;
};
